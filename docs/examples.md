## 4. Factorial

```
(ПОЯСНИТЕЛЬНАЯ-БРИГАДА: Рекурсивная функция вычисления факториала)
ЦИФЕРКА факториал(н: ЦИФЕРКА)
ПОЕХАЛИ
    ЕСЛИ (н <= 1) ТО
        ДРАТУТИ 1;
    ИНАЧЕ
        ДРАТУТИ н * факториал(н - 1);
ФИНАЛОЧКА

(ПОЯСНИТЕЛЬНАЯ-БРИГАДА: Точка входа)
ПРОКРАСТИНИРУЕМ ПОГНАЛИ()
ПОЕХАЛИ
    ЦИФЕРКА число = 0;
    
    ВЫБРОС("Введите число для вычисления факториала:");
    ВБРОС(число);
    
    ЕСЛИ (число < 0) ТО
        ВЫБРОС("Ошибка: число должно быть неотрицательным!");
    ИНАЧЕ ПОЕХАЛИ
        ЦИФЕРКА результат = факториал(число);
        ВЫБРОС("Факториал равен:", результат);
    ФИНАЛОЧКА;
ФИНАЛОЧКА
```

## 5. MultiplicationTable

```
ПРОКРАСТИНИРУЕМ ПОГНАЛИ()
ПОЕХАЛИ
    ЦИФЕРКА н = 0;
    ВЫБРОС("Введите размер таблицы умножения (N):");
    ВБРОС(н);

    (ПОЯСНИТЕЛЬНАЯ-БРИГАДА: Объявляем счетчики заранее, так требует грамматика цикла)
    ЦИФЕРКА и = 1;
    ЦИФЕРКА жи = 1;

    (ПОЯСНИТЕЛЬНАЯ-БРИГАДА: Внешний цикл по строкам)
    ЦИКЛ (и = 1; и <= н; и = и + 1) ПОЕХАЛИ
        
        ВЫБРОС("--- Умножение на ", и, " ---");
        
        (ПОЯСНИТЕЛЬНАЯ-БРИГАДА: Внутренний цикл по столбцам)
        ЦИКЛ (жи = 1; жи <= н; жи = жи + 1) ПОЕХАЛИ
            ЦИФЕРКА произв = и * жи;
            ВЫБРОС(и, " * ", жи, " = ", произв);
        ФИНАЛОЧКА;
        
        ВЫБРОС(""); (ПОЯСНИТЕЛЬНАЯ-БРИГАДА: Пустая строка для разделения)
    ФИНАЛОЧКА;
ФИНАЛОЧКА
```

## 6. Sqrt

```
ПРОКРАСТИНИРУЕМ ПОГНАЛИ()
ПОЕХАЛИ
    ПОЛТОРАШКА число = 0.0;
    
    ВЫБРОС("Введите число для извлечения корня:");
    ВБРОС(число);
    
    ЕСЛИ (число < 0.0) ТО
        ВЫБРОС("Ошибка: нельзя извлечь корень из отрицательного числа");
    ИНАЧЕ ПОЕХАЛИ
        (ПОЯСНИТЕЛЬНАЯ-БРИГАДА: Начальное приближение)
        ПОЛТОРАШКА корень = число;
        ПОЛТОРАШКА точность = 0.0001;
        ПОЛТОРАШКА разница = 1.0;
        
        (ПОЯСНИТЕЛЬНАЯ-БРИГАДА: Итерационный метод Ньютона)
        (ПОЯСНИТЕЛЬНАЯ-БРИГАДА: Вычисляем: x = 0.5 * (x + a / x))
        
        ПОКА (разница > точность) ПОЕХАЛИ
            ПОЛТОРАШКА старыйКорень = корень;
            
            корень = 0.5 * (корень + число / корень);
            
            (ПОЯСНИТЕЛЬНАЯ-БРИГАДА: Проверяем сходимость)
            разница = МОДУЛЬ(старыйКорень - корень);
        ФИНАЛОЧКА;
        
        ВЫБРОС("Корень (приблизительно):", корень);
        ВЫБРОС("Корень * Корень:", корень * корень);
    ФИНАЛОЧКА;
ФИНАЛОЧКА
```

## 7. ReverseString - читает строку, переворачивает её символы и печатает результат

```
ПРОКРАСТИНИРУЕМ ПОГНАЛИ() 
ПОЕХАЛИ
    ЦИТАТА вход = "";
    ВЫБРОС("Введите строку:");
    ВБРОС(вход);

    (ПОЯСНИТЕЛЬНАЯ-БРИГАДА: Используем встроенную функцию ДЛИНА)
    ЦИФЕРКА размер = ДЛИНА(вход);
    
    ЦИТАТА результат = "";
    ЦИФЕРКА и = размер - 1;

    (ПОЯСНИТЕЛЬНАЯ-БРИГАДА: Идем с конца строки к началу)
    ПОКА (и >= 0) 
    ПОЕХАЛИ
        (ПОЯСНИТЕЛЬНАЯ-БРИГАДА: вход[и] возвращает ЦИТАТУ длиной 1)
        результат = результат + вход[и];
        и = и - 1;
    ФИНАЛОЧКА

    ВЫБРОС("Наоборот: ", результат);
ФИНАЛОЧКА
```
